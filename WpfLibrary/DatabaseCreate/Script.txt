CREATE DATABASE LibraryDB;

CREATE TABLE [dbo].[Readers] (
    [Id]               INT           NOT NULL,
    [Year]             INT           NULL,
    [Name]             NVARCHAR (20) NULL,
    [Surname]          NVARCHAR (20) NULL,
    [Patronymic]       NVARCHAR (20) NULL,
    [YearOfBirth]      INT           NULL,
    [Education]        NVARCHAR (20) NULL,
    [Job]              NVARCHAR (20) NULL,
    [Position]         NVARCHAR (20) NULL,
    [WorkPhone]        NVARCHAR (20) NULL,
    [Adress]           NVARCHAR (50) NULL,
    [Phone]            NVARCHAR (20) NULL,
    [PassportData]     NVARCHAR (50) NULL,
    [RegistrationDate] DATE          DEFAULT (getdate()) NULL,
    PRIMARY KEY CLUSTERED ([Id] ASC)
);

CREATE TABLE [dbo].[Books] (
    [InventaryNumber]       INT           NOT NULL,
    [Section]               NVARCHAR (10) NULL,
    [SectionNumber]         NVARCHAR (10) NULL,
    [Author]                NVARCHAR (30) NULL,
    [Name]                  NVARCHAR (30) NOT NULL,
    [AdditionalInformation] NVARCHAR (50) NULL,
    [HavkinaNumber]         NVARCHAR (10) NULL,
    [Annotation]            NVARCHAR (50) NULL,
    [Reader]                INT           NULL,
    [DateOfIssue]           DATE          DEFAULT ('1900-01-01') NULL,
    PRIMARY KEY CLUSTERED ([InventaryNumber] ASC),
    FOREIGN KEY ([Reader]) REFERENCES [dbo].[Readers] ([Id])
);
